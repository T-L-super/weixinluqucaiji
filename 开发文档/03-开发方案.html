3. 技术实现细节

3.1 项目结构

admission-system/ ├── backend/ │ ├── app/ │ │ ├── main.py # FastAPI 入口 │ │ ├── api/ # API 路由 │ │ │ ├── records.py # 录取记录 API │ │ │ ├── tasks.py # 任务管理 API │ │ │ └── stats.py # 统计 API │ │ ├── models/ # 数据库模型 │ │ │ ├── database.py # 数据库连接 │ │ │ ├── record.py # 录取记录模型 │ │ │ └── task.py # 任务模型 │ │ ├── schemas/ # Pydantic 模型 │ │ └── core/ # 核心逻辑 │ │ ├── collector.py # OpenClaw 采集器 │ │ └── extractor.py # 数据提取器 │ └── requirements.txt ├── frontend/ │ ├── src/ │ │ ├── views/ │ │ └── components/ │ └── dist/ # 构建产物 ├── data/ │ └── admission_system.db # SQLite 数据库 ├── Dockerfile └── docker-compose.yml

3.2 数据库配置

# 开发环境（SQLite） DATABASE_URL=sqlite:///./admission_system.db # 生产环境（MySQL） DATABASE_URL=mysql://root:password@localhost/admission_system

3.3 FastAPI 主入口

from fastapi import FastAPI from fastapi.staticfiles import StaticFiles from fastapi.middleware.cors import CORSMiddleware app = FastAPI(title="录取信息管理系统") # CORS 中间件 app.add_middleware( CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"], ) # 静态文件服务（前端） app.mount("/assets", StaticFiles(directory="frontend/dist/assets"), name="assets") # API 路由 from app.api import records, tasks, stats app.include_router(records.router, prefix="/api/records") app.include_router(tasks.router, prefix="/api/tasks") app.include_router(stats.router, prefix="/api/stats") # SPA 路由支持 @app.get("/{full_path:path}") async def serve_spa(full_path: str): return FileResponse("frontend/dist/index.html")
